\documentclass[varwidth = 15cm]{standalone}
\usepackage{ucblalgo}

\begin{document}

\begin{ucblalgo}

  \Algorithme {
    ajouter une suivante nullptr à la cellule sentinelle \;
    positionner un curseur "courante" sur la suivante de la sentinelle au niveau 0 \;
    positionner un curseur "précédente" sur la cellule sentinelle \;
    \Tq{la cellule courante n'a pas atteint la fin de la liste} {
      tirer a pile ou face si la cellule courante est sélectionnée \;
      \Si{elle est sélectionnée} {
        ajouter une suivante nullptr à la cellule courante \;
        la suivante au niveau 1 de la cellule précédente devient la cellule courante \;
        la précédente devient la cellule courante \;
      }
      la cellule courante avance sur sa suivante \;
    }
  }
\end{ucblalgo}

\end{document}
